**Goal:** Add `@react-three/rapier` physics to the 3D graph view — nodes spring on hover, idle drift. Feature-flagged via `NEXT_PUBLIC_GRAPH_PHYSICS`. Zero impact when flag is off.

**Architecture:** Create a physics-enhanced scene variant `scene-physics.tsx` that wraps the existing `scene.tsx` nodes in Rapier `RigidBody`. - [ ] **Step 2: Create physics scene**

```tsx
"use client";

import { useRef } from "react";
import { useFrame } from "@react-three/fiber";
import { Physics, RigidBody, type RapierRigidBody } from "@react-three/rapier";
import { useReducedMotion } from "@/lib/use-reduced-motion";
// Re-use the base scene's graph model but wrap nodes in RigidBody
import { HeroGraphScene } from "./scene";

/**
 * Physics-enhanced hero graph scene.
 * Wraps the existing scene in a Rapier Physics world with zero gravity (space feel).
 * Nodes are kinematic rigid bodies — the force-directed layout drives their position,
 * physics adds micro-impulses on hover for a spring/bounce effect.
 *
 * Only imported when NEXT_PUBLIC_GRAPH_PHYSICS=true — zero bundle impact when off.
 */
export function HeroGraphScenePhysics(props: React.ComponentProps<typeof HeroGraphScene>) {
  return (
    <Physics gravity={[0, 0, 0]} timeStep="vary">
      <HeroGraphSceneWithImpulses {...props} />
    </Physics>
  );
}

function HeroGraphSceneWithImpulses(props: React.ComponentProps<typeof HeroGraphScene>) {
  const prefersReducedMotion = useReducedMotion();
  const bodyRef = useRef<RapierRigidBody>(null);

  // Apply a gentle idle drift — tiny sinusoidal impulse each frame to make the
  // graph feel alive. Suppressed when user prefers reduced motion.
  useFrame(({ clock }) => {
    if (prefersReducedMotion || !bodyRef.current) return;
    const t = clock.elapsedTime;
    bodyRef.current.applyImpulse(
      { x: Math.sin(t * 0.3) * 0.0001, y: Math.cos(t * 0.2) * 0.0001, z: 0 },
      true,
    );
  });

  // The base scene handles rendering — physics wraps it
  return <HeroGraphScene {...props} />;
}
```

NOTE: After reading `scene.tsx`, adapt the above to match the actual prop types and how nodes are structured. If nodes are instanced meshes, each instance needs its own RigidBody — the implementation may need to be more granular. Use the base scene's structure as the guide.
